#include <iostream> 
#include <string>
using namespace std;

int main() {
    unsigned long long binary = 0, rang = 1;
    int decimal = 0;
    cin >> decimal;
    while (decimal > 0) {
        binary += (decimal % 2) * rang;
        decimal /= 2;
        rang *= 10;
    }
    cout << binary;
    return 0;
}
